CREATE TABLE USERS (UserId COUNTER CONSTRAINT PK_UserID PRIMARY KEY, Login VARCHAR(50) NOT NULL, IsAdmin BIT NOT NULL)

CREATE UNIQUE INDEX IX_UsersLogin ON USERS (Login)


CREATE TABLE THEMES (ThemesId COUNTER CONSTRAINT PK_ThemesId PRIMARY KEY, Name VARCHAR(50))

CREATE UNIQUE INDEX IX_NameThemes ON THEMES (Name)

CREATE TABLE QUESTION (QuestionId COUNTER CONSTRAINT PK_QuestionId PRIMARY KEY, ThemesId INT, Desctiption VARCHAR(250))

CREATE TABLE QUESTION (QuestionId COUNTER CONSTRAINT PK_QuestionId PRIMARY KEY, ThemesId INT, Desctiption VARCHAR(250), CONSTRAINT FK_THEMES FOREIGN KEY (ThemesId) REFERENCES THEMES (ThemesId) ON UPDATE CASCADE ON DELETE CASCADE)

CREATE TABLE ANSWERS (AnswerID COUNTER CONSTRAINT PK_AnswerID PRIMARY KEY, QuestionId INT NOT NULL, AnswerText VARCHAR(50) NOT NULL, CorrectAnswer BIT NOT NULL, CONSTRAINT FK_QuestionId FOREIGN KEY (QuestionId) REFERENCES QUESTION (QuestionId) ON UPDATE CASCADE ON DELETE CASCADE)

CREATE TABLE GAMES (DateBegin datetime CONSTRAINT PK_DateTimeGame PRIMARY KEY, UserId int, ThemesId int, DateEnd datetime, CountQuestion int, CountCorrectQuestion int, CONSTRAINT FK_GamesUserId FOREIGN KEY (UserId) REFERENCES Users (UserId) ON UPDATE CASCADE ON DELETE CASCADE, CONSTRAINT FK_GamesThemesId FOREIGN KEY (ThemesId) REFERENCES THEMES (ThemesId) ON UPDATE CASCADE ON DELETE CASCADE)
